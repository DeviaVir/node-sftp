 16 /**
 17  * @class Sftp object
 18  * @extends events.EventEmitter
 19  * 
 20  * @property {Number} state        state the Sftp instance is in
 21  * @property {String} activeBuffer buffer that contains the output of the active Sftp command
 22  * @property {Object} callbacks    hashmap that contains callbacks for all Sftp commands
 23  * @property {Array}  queue        simple FIFO queue for commands to hold until (re)connection
 24  * @property {Object} cmdOptions   list of options per Sftp command, fetched from the output of the help command
 25  * 
 26  * @param {Object}   options                    that contain all the info required 
 27  *                                              to setup a connection
 28  *                                              to a remote server. Structure:
 29  * @param {String}   options.username           Username on remote host.
 30  * @param {String}   [options.password]         Password on remote host, not required if private key is provided.
 31  * @param {String}   [options.privateKey]       Private key for username, not required if password is provided.
 32  * @param {String}   [options.host="localhost"] Hostname or IP to connect to. Default: 'localhost'
 33  * @param {Number}   [options.port=22]          SSH Port number. Default: 22
 34  * @param {Number}   [options.timeout=10000]    Inactivity timeout in milliseconds. Default: 10000ms
 35  * @param {Boolean}  [options.autoconnect=true] Connect on instantiation. Default: true
 36  * @param {String}   [options.exec]             Command to be executed right after a connection is established.
 37  * @param {Function} [cbconnect]                callback to invoke right after a connection is established.
 38  * @type  {Sftp}
 39  */
 40 function Sftp(options, cbconnect) {
 41     Events.EventEmitter.call(this);
 42 
 43     this.options = Util.extend({
 44         host: "localhost",
 45         port: 22,
 46         timeout: 10000,
 47         autoconnect: true
 48     }, options || {});
 49     
 50     // initialize state variables
 51     this.state           = 0;
 52     this.activeCmdBuffer = "";
 53     this.callbacks       = {};
 54     this.queue           = [];
 55     this.cmdOptions      = {};
 56     
 57     var o     = this.options,
 58         _self = this;
 59     // plaintext private key needs to be written to file first
 60     if (o.privateKey && o.privateKey.indexOf("BEGIN RSA PRIVATE KEY") > 0) {
 61         var _self = this;
 62         Ssh.writeKeyFile(o.privateKey, function(err, file) {
 63             if (err)
 64                 return cbconnect(err);
 65             o.privateKey = file;
 66             _self.$privateKeyTemp = file;
 67             afterInit();
 68         });
 69     }
 70     else
 71         afterInit();
 72 
 73     function afterInit() {
 74         _self.state = Sftp.STATE_DISCONNECTED;
 75         _self.emit("ready");
 76         cbconnect = cbconnect || o.callback || K;
 77 
 78         if (o.exec) {
 79             var args = o.exec.split(" "),
 80                 func = parts.shift(),
 81                 cb   = o.callback || cbconnect;
 82             if (!_self[func])
 83                 return cb("Unsupported method '" + func + "' specified in the exec option");
 84             _self.connect(cbconnect);
 85             args.push(cb);
 86             _self[func].apply(_self, args);
 87         }
 88         else if (_self.queue.length) {
 89             _self.connect(cbconnect);
 90             _self.exec.apply(_self, _self.queue.shift());
 91         }
 92         else if (o.autoconnect) {
 93             _self.connect(cbconnect);
 94         }
 95     }
 96 }

144     /**
145      * Setup an Sftp connection to a remote host
146      * 
147      * @param {Function} cbconnect callback to invoke right after a connection is established.
148      * @type  {void}
149      */
150     this.connect = function(cbconnect) {
151         if (!(this.state & Sftp.STATE_DISCONNECTED)) {
152             if (cbconnect) {
153                 cbconnect(this.state & Sftp.STATE_CONNECTED 
154                     ? null 
155                     : this.state & Sftp.STATE_CONNECTING
156                         ? "SFTP Error: already connecting to a host, please be patient" 
157                         : "SFTP Error: invalid state."
158                 );
159             }
160             return;
161         }
162 
163         this.state = Sftp.STATE_CONNECTING;
164 
165         var o     = this.options,
166             args  = ["-o", "Port=" + o.port],
167             _self = this;
168         if (o.privateKey)
169             args = args.concat(Ssh.buildArgs(o.privateKey));
170         args.push(
171             // first we push the correct hostname (appended with the path, if supplied)
172             (o.username ? o.username + "@" : "") + o.host + (o.home ? ":" + o.home : "")
173         );
174         // push the connection string as argument:
175         console.log("launching: sftp " + args.join(" "));
176         var ps = Tty.open("sftp", args);
177         this.socket = ps[0];
178         this.child = ps[1];
179 
180         this.socket.on("data", function(data) {
181             if (_self.state & Sftp.STATE_DISCONNECTED)
182                 _self.socket.destroy();
183             parseReply.call(_self, data.toString(), 1);
184         });
185 
186         this.child.on("exit", function(code) {
187             _self.emit("disconnect", code);
188             _self.socket.flush();
189             _self.state = Sftp.STATE_DISCONNECTED;
190             if (_self.$privateKeyTemp)
191                 Fs.unlink(_self.$privateKeyTemp);
192         });
193 
194         this.callbacks["connect"] = function() {
195             _self.state = Sftp.STATE_CONNECTED;
196 
197             send.call(_self, "help", "help", function(lines) {
198                 registerFeatures.call(_self, lines);
199                 _self.emit("connect");
200                 cbconnect && cbconnect();
201 
202                 if (_self.queue.length)
203                     _self.exec.apply(_self, _self.queue.shift());
204             }, K);
205         };
206     };

208     /**
209      * Parse the output of a 'help' or '?' command and fetch the command options
210      * that are supported by the remote server.
211      * The supported options are cached and used by functions like ls()
212      * 
213      * @param {Array} lines Array of lines that contain strings of terminal output
214      * @type  {void}
215      * @private
216      */
217     function registerFeatures(lines) {
218         lines = lines.slice(lines.indexOf("Available commands:") + 1);
219         var _self = this;
220         lines.forEach(function(line) {
221             // parse the help output for a command
222             var m = line.match(/^([^\t]*)[\s]{2,}([^\t]+)$/);
223             if (!m || m.length != 3)
224                 return;
225             // remove the unnecessary trailing spaces
226             m[1] = m[1].replace(/[\s]+$/, "");
227             if (!m[1])
228                 return;
229             // parse the command structure to fetch the command options.
230             // the regex is more generic then it must be, because I might use it
231             // for something else later...
232             m = m[1].match(/([^\s]*)(?:[\s]+)?([^\s]*)?(?:[\s]+)?([^\s]*)?(?:[\s]+)?([^\s]*)?/);
233             if (m[2] && m[2].substr(0, 2) == "[-")
234                 _self.cmdOptions[m[1]] = m[2].substr(2, m[2].length - 3).split("");
235         });
236     }

357     /**
358      * Small and simple routine to transform a collection of output strings into
359      * an object of property for each filesystem node.
360      * 
361      * @param {Array} lines
362      * @type  {Object}
363      * @private
364      */
365     function parseListing(lines) {
366         var res = [];
367         lines.forEach(function(line) {
368             if (!line || /\/[\.]{1,2}$/.test(line))
369                 return;
370             var m = line.match(/^[\s]*([drwx-]+)[\s]*([\d]+)[\s]*([\d]+)[\s]*([\d]+)[\s]*([\d]+)[\s]*([\w\s\d]*)([\d]{2}\:[\d]{2})[\s]*(.*)$/);
371             if (!m)
372                 return;
373             res.push({
374                 perms: m[1],
375                 uid: parseInt(m[3]),
376                 gid: parseInt(m[4]),
377                 size: parseInt(m[5]),
378                 date: m[6],
379                 time: m[7],
380                 path: m[8]
381             });
382         });
383         return res;
384     }

386     /**
387      * Execute the 'ls' command, which is used for {@link Sftp#readdir}, {@link Sftp#stat}, 
388      * {@link Sftp#fstat} and {@link Sftp#lstat} respectively. For {@link Sftp#stat} 
389      * and its siblings, using 'ls' is the only way to retrieve a filesystem node's
390      * stats.
391      * Important: 'ls' output results are cached for 10 seconds, to improve performance.
392      * 
393      * @param {String}   path
394      * @param {Function} cbls
395      * @type  {void}
396      * @private
397      */
398     function ls(path, cbls) {
399         if (!this.$lsCache)
400             this.$lsCache = {};
401         var cache = this.$lsCache[path],
402             now   = Date.now(),
403             _self = this;
404         if (cache) {
405             if (cache.expires >= now)
406                 return cbls(null, cache.result);
407             else
408                 delete this.$lsCache[path];
409         }
410 
411         var cmd = "ls -l";
412         if (this.cmdOptions["ls"].indexOf("a") > -1)
413             cmd += "a";
414         if (this.cmdOptions["ls"].indexOf("n") > -1)
415             cmd += "n";
416         this.exec("ls", cmd + " " + (path || ""), function(lines) {
417             var err = isError(lines),
418                 res = parseListing(lines);
419             if (!err) {
420                 _self.$lsCache[path] = {
421                     expires: now + 10000,
422                     result: res
423                 };
424             }
425             cbls(err, res);
426         });
427     }

516     /**
517      * Helper function to get an octal representation from terminal output in a 
518      * human readable form.
519      * Example: 'drwxr-xr-x' --> '755' or '-rw-rw-r--' --> '664'
520      * 
521      * @param {String} rwx
522      * @type  {String}
523      * @private
524      */
525     function getOct(rwx) {
526         var oct = "";
527         for (var c, o = 0, i = 0, l = rwx.length; i < l; ++i) {
528             c = rwx.charAt(i);
529             if (i % 3 === 0) {
530                 oct += "" + o;
531                 o = 0;
532             }
533             c = rwx.charAt(i + 1);
534             o += c == "r" ? 4 : c == "w" ? 2 : c == "x" ? 1 : 0;
535         }
536         return oct;
537     }

611     /**
612      * Asynchronous lstat(2). The callback gets two arguments (err, stats) where 
613      * stats is a sftp.Stats object. lstat() is identical to stat(), except that if 
614      * path is a symbolic link, then the link itself is stat-ed, not the file that 
615      * it refers to.
616      * 
617      * @param {String}   path
618      * @param {Function} callback
619      * @type  {void}
620      */
621     this.stat  = function(path, callback) {
622         var _self = this,
623             parts = path.split("/"),
624             node  = parts.pop(),
625             root  = parts.join("/");
626         if (root.charAt(0) != "/") {
627             this.pwd(function(err, pwd) {
628                 if (err)
629                     return callback(err);
630                 pwd = pwd.replace(/[\/]+$/, "");
631                 root = pwd + "/" + root.replace(/^[\/]+/, "");
632                 afterPwd();
633             });
634         }
635         else
636             afterPwd();
637         
638         function afterPwd() {
639             ls.call(_self, root, function(err, list) {
640                 list = list.filter(function(item) {
641                     return item.path.split("/").pop() === node;
642                 });
643                 if (list.length === 0)
644                     return callback("Couldn't stat remote file: No such file or directory");
645                 return callback(null, new Stat(list[0]));
646             });
647         }
648     };

703      * 
704      * @param {String}   type
705      * @param {String}   cmd
706      * @param {Function} cbexec
707      * @param {Function} cbprogress
708      * @type  {void}

710     this.exec = function(type, cmd, cbexec, cbprogress) {
711         var conn = this.state & Sftp.STATE_CONNECTED;

721     /**
722      * Send an Sftp command to the remote host, i.e. write it to the CLI stream
723      * 
724      * @param {String}   type
725      * @param {String}   cmd
726      * @param {Function} cbsend
727      * @param {Function} cbprogress
728      * @type  {void}
729      * @private
730      */
731     function send(type, cmd, cbsend, cbprogress) {
732         this.activeCmd = type;
733         this.activeCmdBuffer = "";
734         if (cbprogress && cbsend) {
735             this.callbacks[type] = cbsend;
736             this.callbacks[type + "_progress"] = cbprogress;
737         }
738         this.socket.write(new Buffer(cmd + "\r\n"));
739         this.socket.resume();
740     }
741     
742     /**
743      * Parse the STDOUT of the sftp child process, collect the lines of data for
744      * the active command and detect if the active command has finished executing.
745      * If the active command is done, respective callbacks are called.
746      * 
747      * @param {String} data
748      * @type  {void}
749      * @private
750      */
751     function parseReply(data) {
752         //console.log("data: ", data, data.split("\n").length + " parts");
753         this.emit("data", data);
754         if (!this.activeCmd && !(this.state & Sftp.STATE_CONNECTING))
755             return;
756         var cbdone, cbprogress;
757         if (data.indexOf("sftp>") > -1 || (this.activeCmd == "bye" && data.indexOf("bye") > -1)) {
758             if (this.state & Sftp.STATE_CONNECTING && this.callbacks["connect"]) {
759                 this.callbacks["connect"]();
760                 delete this.callbacks["connect"];
761             }
762             // check if a command has finished executing:
763             else if (cbdone = this.callbacks[this.activeCmd]) {
764                 delete this.callbacks[this.activeCmd];
765                 delete this.callbacks[this.activeCmd + "_progress"];
766                 this.activeCmd = null;
767                 cbdone((this.activeCmdBuffer + data).split(/[\n\r]+/).filter(function(line) {
768                     return line.indexOf("sftp>") === -1;
769                 }));
770                 this.activeCmdBuffer = "";
771             }
772             if (!this.activeCmd && this.queue.length && this.state & Sftp.STATE_CONNECTED)
773                 this.exec.apply(this, this.queue.shift());
774         }
775         else if (cbprogress = this.callbacks[this.activeCmd + "_progress"]) {
776             this.activeCmdBuffer += data;
777             cbprogress(data);
778         }
779     }

781     /**
782      * Detect an error reply from a collection of output lines.
783      * 
784      * @param {Array} lines
785      * @private
786      */
787     function isError(lines) {
788         var err = null;
789         lines.forEach(function(line) {
790             if (line.indexOf("No such file or directory") > -1)
791                 err = line;
792         });
793         return err;
794     }
